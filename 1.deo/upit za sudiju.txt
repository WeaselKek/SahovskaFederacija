DECLARE
majs1 VARCHAR(30);
majs2 VARCHAR(30);
orga1 VARCHAR(30);
orga2 VARCHAR(30);
part PARTIJA.ID%TYPE:=10001;
BEGIN

BEGIN
SELECT SAHISTA.IME,SAHISTA.PREZIME
INTO majs1,majs2
FROM PARTIJA, SUDIJA, MAJSTOR, SAHISTA
WHERE PARTIJA.SUDIJA_ID = SUDIJA.SUDIJA_ID AND SUDIJA.SUDIJA_ID = MAJSTOR.SUDIJA_ID AND MAJSTOR.REG_BR = SAHISTA.REG_BR AND PARTIJA.ID=part;
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        majs1 := NULL;
        majs2 := NULL;
END;

BEGIN
SELECT ORGANIZATOR.IME,ORGANIZATOR.PREZIME
INTO orga1,orga2
FROM PARTIJA, SUDIJA, ORGANIZATOR
WHERE PARTIJA.SUDIJA_ID = SUDIJA.SUDIJA_ID AND SUDIJA.SUDIJA_ID = ORGANIZATOR.SUDIJA_ID  AND PARTIJA.ID=part;
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        orga1 := NULL;
        orga2 := NULL;
END;

IF majs1 IS NULL
THEN
DBMS_OUTPUT.PUT_LINE('sudija je ' || orga1 || ' '|| orga2);
ELSE
DBMS_OUTPUT.PUT_LINE('sudija je ' || majs1 || ' '|| majs2);
END IF;

END;
